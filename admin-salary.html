Google Sheets — live data source

Maintain your salary database in a Google Sheet and sync it here automatically. No coding needed.

  1. Create a Google Sheet with columns: id, company, position, sector, country, years_exp, salary, unpaid, year
  2. Go to Extensions → Apps Script and paste the doGet() script below
  3. Deploy as Web App → access: Anyone → copy the URL
  4. Paste the URL into the field below and click Sync from Sheet
function doGet() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); const rows = sheet.getDataRange().getValues(); const keys = rows[0]; const data = rows.slice(1).map(r => { const obj = {}; keys.forEach((k, i) => obj[k] = r[i]); obj.salary = Number(obj.salary) || 0; obj.unpaid = String(obj.unpaid).toLowerCase() === 'true'; return obj; }); return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
